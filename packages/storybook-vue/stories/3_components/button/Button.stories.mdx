import { Meta, Story, Props, Preview, Description } from '@storybook/addon-docs/blocks';
import ScaleButton from "./ScaleButton.vue";

<Meta
  title='Components/Button'
  component={ScaleButton}
  argTypes={{
    variant: {
      control: { type: 'select', options: ['primary', 'secondary'] }
    },
    label: {
      defaultValue: 'Label',
      control: { type: 'text' }
    },
    disabled: {
      control: { type: 'boolean' }
    }
  }}
/>

export const Template = (args, { argTypes }) => ({
  components: { ScaleButton },
  props: {
    label: String,
    icon: String,
    ...ScaleButton.props
  },
  template: `
    <scale-button
      :title="title"
      :disabled="disabled"
      :size="size"
      :href="href"
      :target="target"
      :variant="variant"
      :icon-only="iconOnly"
      @click="action('clicked')"
    >
      <scale-icon-action-search v-if="icon === 'before'" :size="size == 'small' ? 16 : 24" />
      {{ iconOnly !== true ? label : null }}
      <scale-icon-action-add v-if="iconOnly" :size="size == 'small' ? 16 : 24" />
      <scale-icon-navigation-right v-if="icon === 'after'" :size="size == 'small' ? 16 : 24" />
    </scale-button>
  `
})
